
--ver supervisores en sylabus
SELECT 
    ss.ID_Syllabus,
    sup.name AS supervisor,
    sig.name AS materia
FROM Syllabus_supervisor ss
JOIN Supervisor sup ON ss.ID_supervisor = sup.ID_supervisor
JOIN Syllabus sig ON ss.Syllabus_id = sig.Syllabus_id;
-ver programa asoiado

SELECT 
    pr.ID_program,
    pr.content,
    pr.hours,
    pr.semester,
    pr.school,
    sig.name AS materia
FROM Program pr
JOIN Syllabus sig ON pr.Syllabus_id = sig.Syllabus_id;

SELECT 
    s.ID_Syllabus,
    s.academic_year,
    p.name AS profesor,
    sig.name AS materia,
    s.upload_date,
    s.parallel_code,
    s.is_active
FROM Syllabus s
JOIN Professor p ON s.professor_id = p.professor_id
JOIN Syllabus sig ON s.Syllabus_id = sig.Syllabus_id;
